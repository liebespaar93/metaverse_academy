템플릿을 사용하여 보다 유연하게 요소들을 사용할 수 있다

```cpp filename="warrior.hpp"
#include <iostream>
#ifndef WARRIOR_HPP
#define WARRIOR_HPP

#include "soldier.hpp"

template <typename T>
class Warrior : public Soldier
{

private:
    T damagePower;

public:
    Warrior(const char *text, int level, int speed, int damage) : Soldier(text, level, speed)
    {
        this->damagePower = damage;
    }

    Warrior(const Warrior &ref) : Soldier(ref.name, ref.level, ref.speed)
    {
    }

    ~Warrior() {}

    Warrior &operator=(const Warrior &ref) { return *this; }

    void ShowName()
    {
        std::cout << "name : Warrior" << std::endl;
    }
};

#endif
```

```cpp filename="main.cpp"
#include "soldier.hpp"
#include "warrior.hpp"
#include "archer.hpp"

#include <iostream>

int main()
{
    Warrior<int> *myWarrior = new Warrior<int>("전사",100,100,100);
    std::cout << myWarrior << std::endl;
    delete(myWarrior);
    return 0;
}
```

상위 코드처럼 타입을 임의의 변수로 받아 필요한 타입으로 정의 하는 것이 가능하다.\
이러한 장점을 이용하여 보다 범용적인 클래스 및 함수를 만들어 사용한다.
